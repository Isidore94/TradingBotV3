in tradingview this is how i currently use pinescript to caculate the levels: //@version=5
indicator("Highs of High Relative Volume Candle", overlay=true)

// Calculate the 50-day moving average of volume
avgVolume50 = ta.sma(volume, 50)

// Calculate relative volume
relVolume = volume / avgVolume50

// Conditions to check relative volume levels
high_rel_volume_2_3 = relVolume >= 2 and relVolume < 3
high_rel_volume_above_3 = relVolume >= 3

// Retrieve the high of the last high relative volume candle
high_level_2_3 = ta.valuewhen(high_rel_volume_2_3, high, 0)
high_level_above_3 = ta.valuewhen(high_rel_volume_above_3, high, 0)

// Plot the high levels as lines
if (high_rel_volume_2_3)
    line.new(bar_index, high_level_2_3, bar_index - 1, high_level_2_3, width=1, color=color.red, extend=extend.left)

if (high_rel_volume_above_3)
    line.new(bar_index, high_level_above_3, bar_index - 1, high_level_above_3, width=1, color=color.green, extend=extend.left)

6. additionally I like to use cloud lines as S/R levels heres my tradinview script to generate the ones i care about 
//@version=5
indicator("Leading Span B Lines", shorttitle="LSB-Lines", overlay=true)

basePeriods = input.int(26, minval=1, title="Base Line Length")
laggingSpan2Periods = input.int(52, minval=1, title="Leading Span B Length")
displacement = input.int(26, minval=1, title="Displacement")

// Donchian channel function
donchian(len) => math.avg(ta.lowest(len), ta.highest(len))

// Calculation of Leading Span B
leadLine2 = donchian(laggingSpan2Periods)

// Plot Leading Span B
p2 = plot(leadLine2, offset = displacement, color=color.new(#EF9A9A, 0), title="Leading Span B")

// Function to check if a line is flat
isFlat(line) => ta.sma(line, 8) == line

// Detecting flat condition for Leading Span B
flatCloudCondition2 = isFlat(leadLine2)

var line flatLine2 = na
if flatCloudCondition2
    if na(flatLine2)
        flatLine2 := line.new(bar_index[1], leadLine2, bar_index, leadLine2, width=1, color=color.purple, extend=extend.right)
else
    if not na(flatLine2)
        line.set_x2(flatLine2, bar_index[1])
        flatLine2 := na
